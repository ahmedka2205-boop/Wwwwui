#!/usr/bin/env python3 # server.py """ Local integrated simulation server - Serves a client-side A/B simulation UI (embedded HTML/JS) - Accepts uploaded simulation JSON files (POST /upload) - Processes uploaded sessions automatically to CSV summary - Serves uploaded and processed files for download Security / usage notes: - Binds to 127.0.0.1 only (localhost). Do NOT expose to public networks. - Basic Auth protects the server UI and upload/download endpoints. - Intended for SAFE synthetic simulations on devices/accounts you control. - Change SIM_ADMIN_USER and SIM_ADMIN_PASS via environment variables. Requirements: pip install flask werkzeug Run: SIM_ADMIN_USER=admin SIM_ADMIN_PASS=changeme python server.py Then open: http://127.0.0.1:5000 """ import os import pathlib import json import time import base64 import statistics from flask import Flask, request, jsonify, send_from_directory, abort, Response, make_response # ------------------------- # Configuration # ------------------------- BASE_DIR = pathlib.Path(__file__).parent.resolve() UPLOAD_DIR = BASE_DIR / "uploads" PROCESSED_DIR = BASE_DIR / "processed" UPLOAD_DIR.mkdir(exist_ok=True) PROCESSED_DIR.mkdir(exist_ok=True) # Basic auth defaults (override via environment variables) ADMIN_USER = os.environ.get("SIM_ADMIN_USER", "admin") ADMIN_PASS = os.environ.get("SIM_ADMIN_PASS", "changeme") ALLOWED_EXT = {".json"} MAX_UPLOAD_MB = 20 # ------------------------- # Helper: Basic Auth # ------------------------- def check_auth_header(auth_header): """ Expecting header like: 'Basic base64(user:pass)' """ if not auth_header: return False try: scheme, credentials = auth_header.split(None, 1) if scheme.lower() != "basic": return False decoded = base64.b64decode(credentials).decode("utf8", errors="ignore") user, pw = decoded.split(":", 1) return user == ADMIN_USER and pw == ADMIN_PASS except Exception: return False def require_auth(): return Response("Authentication required", 401, {"WWW-Authenticate": 'Basic realm="LocalSim"'}) # ------------------------- # Analysis functions (integrated from analyze_simulation.py) # ------------------------- def compute_from_sessions(sessions): """ Compute aggregated rows per variant: - impressions, clicks, submissions, CTR, form_completion_rate, median_time_on_page, mean_time_on_page Returns list of dict rows. """ agg = {} for s in sessions: v = s.get("variant", "unknown") agg.setdefault(v, {"impressions": 0, "clicks": 0, "submissions": 0, "times": []}) agg[v]["impressions"] += 1 if s.get("metrics", {}).get("clicked"): agg[v]["clicks"] += 1 if s.get("metrics", {}).get("submitted"): agg[v]["submissions"] += 1 t = s.get("metrics", {}).get("time_on_page_sec") if t is not None: try: agg[v]["times"].append(float(t)) except Exception: pass rows = [] for v, d in agg.items(): times = d["times"] median = statistics.median(times) if times else 0 mean = statistics.mean(times) if times else 0 ctr = (d["clicks"] / d["impressions"]) if d["impressions"] else 0 form_completion = (d["submissions"] / d["clicks"]) if d["clicks"] else 0 rows.append({ "variant": v, "impressions": d["impressions"], "clicks": d["clicks"], "submissions": d["submissions"], "CTR": ctr, "form_completion_rate": form_completion, "median_time_on_page": median, "mean_time_on_page": mean }) return rows def write_csv(rows, filename): import csv if not rows: raise ValueError("No rows to write") keys = list(rows[0].keys()) with open(filename, "w", newline="", encoding="utf8") as f: writer = csv.DictWriter(f, fieldnames=keys) writer.writeheader() for r in rows: writer.writerow(r) return filename def analyze_sessions_file(input_path, out_dir=None): """ Read a sessions JSON (structure: {sessions: [...]}) and create a CSV summary in out_dir. Returns path to csv file. """ input_path = pathlib.Path(input_path) out_dir = pathlib.Path(out_dir or input_path.parent) with open(input_path, "r", encoding="utf8") as f: obj = json.load(f) sessions = obj.get("sessions") or obj.get("session_pool") or [] if not sessions: raise ValueError("No session records found in input") rows = compute_from_sessions(sessions) out_csv = out_dir / (input_path.stem + "_summary.csv") write_csv(rows, str(out_csv)) return str(out_csv) # ------------------------- # Flask app # ------------------------- app = Flask(__name__) app.config["MAX_CONTENT_LENGTH"] = MAX_UPLOAD_MB * 1024 * 1024 def is_allowed(filename): return pathlib.Path(filename).suffix.lower() in ALLOWED_EXT # Embedded HTML page (client simulation). Kept self-contained; connects to /upload on same origin. # Note: this page is intentionally restricted (served from localhost). It performs client-side A/B simulation, # allows export of sessions JSON and can POST it to /upload. CLIENT_HTML = r""" محاكاة آمنة A/B - Local Sim

محاكاة آمنة A/B: رابط → صفحة هبوط → مسح QR / إدخال بيانات

تشغيل محلي فقط — بيانات صناعية — لا تُرسل إلى أي جهة خارج جهازك ما لم ترفعها إلى خادمك المحلي.

1) موافقة وتشغيل الوضع الآمن

2) إعداد A/B

عنصرVariant AVariant B
عنوان الإعلان
وصف الإعلان
احتمال النقر (0-1)
احتمال إرسال النموذج بعد النقر

النتائج المجمعة (مختصر)

يمكن تنزيل التفاصيل بصيغة JSON وتحليلها محلياً أو رفعها إلى خادمك المحلي.
ملاحظة: الرفع يتطلب أن تشغّل server.py على جهازك وأن تكون لديك بيانات المصادقة.

سجل الأحداث التجريبي (محلي، صناعي)

لا توجد أحداث بعد.
ملاحظة قانونية: هذه الأداة لأغراض البحث والمحاكاة فقط. لا تُستخدم لاستهداف أو اختراق أي فرد أو حساب حقيقي.
""" # ------------------------- # Routes # ------------------------- @app.route("/", methods=["GET"]) def ui_root(): # Serve the HTML client page (no auth required to GET the client so researcher can open it), # but uploads and processed endpoints are protected. # We still print note for safety. resp = make_response(CLIENT_HTML) # set CSP header again resp.headers["Content-Security-Policy"] = "default-src 'self'; connect-src 'self'; img-src 'self' data:; script-src 'self'; style-src 'self' 'unsafe-inline'" return resp @app.route("/upload", methods=["POST"]) def upload(): # Protect upload endpoint with basic auth auth = request.headers.get("Authorization") if not check_auth_header(auth): return require_auth() if "file" not in request.files: abort(400, "no file part") f = request.files["file"] if f.filename == "": abort(400, "no selected file") filename = pathlib.Path(f.filename).name if not is_allowed(filename): abort(400, "invalid extension") timestamp = int(time.time()) dest = UPLOAD_DIR / f"{timestamp}_{filename}" f.save(dest) # process file (analyze) immediately try: csvpath = analyze_sessions_file(str(dest), out_dir=str(PROCESSED_DIR)) summary = {"filename": dest.name, "received_at": time.ctime(), "processed_csv": pathlib.Path(csvpath).name} out_json = PROCESSED_DIR / (dest.name + ".summary.json") with open(out_json, "w", encoding="utf8") as ou: json.dump(summary, ou, indent=2, ensure_ascii=False) return jsonify({"status": "ok", "summary": summary}) except Exception as e: return jsonify({"status": "error", "error": str(e)}), 500 @app.route("/uploads/") def get_upload(fname): auth = request.headers.get("Authorization") if not check_auth_header(auth): return require_auth() return send_from_directory(str(UPLOAD_DIR), fname, as_attachment=True) @app.route("/processed/") def get_processed(fname): auth = request.headers.get("Authorization") if not check_auth_header(auth): return require_auth() return send_from_directory(str(PROCESSED_DIR), fname, as_attachment=True) # ------------------------- # CLI launcher # ------------------------- if __name__ == "__main__": print("Local Simulation Server") print("Admin user:", ADMIN_USER) print("Start server on http://127.0.0.1:5000") print("Note: change SIM_ADMIN_USER / SIM_ADMIN_PASS env vars to secure credentials") app.run(host="127.0.0.1", port=5000, debug=False)
